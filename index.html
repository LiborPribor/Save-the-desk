canvas.addEventListener('touchend', () => { if(!isDrawing || playerTrace.length < 10) return; isDrawing = false; const normPlayer = normalize(playerTrace); const normTarget = targetShape; // Předloha je už v měřítku 0-100 let totalDist = 0; let samples = normPlayer.length; // Pro každý hráčův bod najdeme nejbližší bod na předloze normPlayer.forEach(p => { let minDist = Infinity; normTarget.forEach(t => { let d = Math.hypot(p.x - t.x, p.y - t.y); if (d < minDist) minDist = d; }); totalDist += minDist; }); let avgDist = totalDist / samples; // Vyladěná konstanta pro lidštější hodnocení (5.5 je teď férovější) let finalScore = Math.round(Math.max(0, 100 - (avgDist * 5.5))); // Pokud je skóre přes 97, zaokrouhlíme na 100 (bonus za pevnou ruku) if (finalScore > 97) finalScore = 100; drawGhost(playerTrace); animateScore(finalScore); // Uložení db.ref(`users/${userId}/history/${todayStr}`).push(finalScore); if(finalScore > bestEver) db.ref(`users/${userId}/bestScore`).set(finalScore); setTimeout(() => { ctx.clearRect(0,0,canvas.width, canvas.height); document.getElementById('score-display').classList.remove('show'); }, 3500); }); function normalize(pts) { let minX = Math.min(...pts.map(p=>p.x)), maxX = Math.max(...pts.map(p=>p.x)); let minY = Math.min(...pts.map(p=>p.y)), maxY = Math.max(...pts.map(p=>p.y)); let width = maxX - minX; let height = maxY - minY; let size = Math.max(width, height, 1); // Vycentrování a škálování na mřížku 10-90 (stejně jako má targetShape) return pts.map(p => ({ x: ((p.x - minX) / size) * 80 + 10, y: ((p.y - minY) / size) * 80 + 10 })); }
